int valor = 100;

    case 100:
        title = "Junior Associate";

    case 200:
        title = "Senior Associate";
        break;
    case 300:
        title = "Manager";
        break;
    case 400:
        title = "Senior Manager";
        break;

        title = "Associate";
        break;
}
